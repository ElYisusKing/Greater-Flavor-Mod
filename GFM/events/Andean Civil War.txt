#GCO AGGRESSOR
country_event = { 
	id = 61000
	title = "Peru_rejects_GCO"
	desc = "Peru_rejects_GCO"
	major = yes
	is_triggered_only = yes
	
	option = { 
	    name = "We begrudgingly accept South Andean sovereignty."
	    prestige = -15
	    badboy = -2
	    relation = { who = PEU value = -100 }
	    relation = { who = BOL value = -100 }
	    influence = { who = PEU value = -400 }
	    influence = { who = BOL value = -400 }
	    set_country_flag = GCO_backs_down
	}
	
	option = { 
	    name = "This invitation was not optional. Time to put them down!"
		GCO = { add_country_flag = andean_civil_war }
		PEU = {
			all_core = { add_core = GCO }
			set_country_flag = andean_civil_war
			war = {
				target = GCO
				attacker_goal = { casus_belli = cut_down_to_size_no_infamy }
				defender_goal = { casus_belli = cut_down_to_size_no_infamy }
				call_ally = yes
			}
			create_alliance = BOL
			war = { attacker_goal = { casus_belli = call_allies_cb } call_ally = yes } 
		}
		BOL = { set_country_flag = andean_civil_war all_core = { add_core = GCO } }
		badboy = 15 
	}
}

country_event = { 
	id = 61001
	title = "Chile_joins_Peru_against_GCO"
	desc = "Chile_joins_Peru_against_GCO"
	fire_only_once =  yes
	
	trigger = {
		tag = GCO
	    AND = { 
		    PEU = { national_provinces_occupied = 0.70 }
		    GCO = { has_country_flag = andean_civil_war }
		    CHL = { exists = yes } 
	    }
	}

	mean_time_to_happen = { months = 1 }
	
	option = { 
	name = "How convenient. The Chileans will be destroyed along with the rest."
		prestige = 5
		infamy = 1
		GCO = { set_country_flag = Chile_joins_Andean_Civil_War influence = { who = CHL value = -400 } war_exhaustion = 10 }
		BOL = { military_access = CHL }
		PEU = { military_access = CHL }
		CHL = {
			all_core = { add_core = GCO }
			set_country_flag = andean_civil_war
			war = {
				target = GCO
				attacker_goal = { casus_belli = cut_down_to_size_no_infamy }
				defender_goal = { casus_belli = cut_down_to_size_no_infamy }
				call_ally = yes
			}
			create_alliance = PEU
			create_alliance = BOL
			create_allliance = ARG
		}
    }
}
